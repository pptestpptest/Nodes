import <contextTestsImport1>
import <contextTestsImport2>

final class <nodeName>ContextTests: XCTestCase {

    private var analyticsMock: <nodeName>AnalyticsMock!

    @MainActor
    override func setUp() {
        super.setUp()
        analyticsMock = <nodeName>AnalyticsMock()
    }

    @MainActor
    override func tearDown() {
        analyticsMock = nil
        super.tearDown()
    }

    @MainActor
    func testNotNil() {
        let context: <nodeName>ContextImp = givenContext()
        expect(context) != nil
    }

    @MainActor
    func testWorkersIsEmpty() {
        let context: <nodeName>ContextImp = givenContext()
        expect(context.workers).to(beEmpty())
    }

    @MainActor
    func testActivation() {
        let context: <nodeName>ContextImp = givenContext()
        expect(context.isActive) == false
        context.activate()
        expect(context.isActive) == true
        context.deactivate()
        expect(context.isActive) == false
    }

    @MainActor
    private func givenContext(
        workers: [Worker] = []
    ) -> <nodeName>ContextImp {
        <nodeName>ContextImp(
            workers: workers,
            analytics: analyticsMock
        )
    }
}
